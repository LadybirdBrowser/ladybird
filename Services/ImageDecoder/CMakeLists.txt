set(CMAKE_AUTOMOC OFF)
set(CMAKE_AUTORCC OFF)

set(SOURCES
    ConnectionFromClient.cpp
)

    add_library(imagedecoderservice SHARED

        ${SOURCES}

    add_library(imagedecoderservice STATIC ${SOURCES})
endif()
